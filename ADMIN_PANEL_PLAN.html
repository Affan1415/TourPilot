Why React Native + Expo Over Next.js?

❌
Next.js Limitation

Next.js is web-only. You cannot compile Next.js to run on Android or iOS. You would need to maintain 3 separate codebases.

✅
Expo Solution

Expo compiles one React codebase to Web (React DOM), Android (native), and iOS (native). Write once, deploy everywhere.

⚡
Native Performance

Unlike Ionic (WebView), React Native renders actual native components. Your app feels native because it IS native.

🔄
Same React Skills

You already know React. React Native uses the same concepts: components, hooks, state, props. Minimal learning curve.

Why Supabase Direct (No Custom API)?

// Traditional approach: Custom API server needed // Mobile App → Your API Server → Database // You must build, deploy, and maintain the API server // Supabase approach: Direct connection // Mobile App → Supabase (with RLS) → Database // No API server to maintain! import { createClient } from '@supabase/supabase-js' const supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY) // Same code works on Web, Android, and iOS! const { data: bookings } = await supabase .from('bookings') .select('*, customer:customers(*), tour:tours(*)') .eq('status', 'confirmed') // Row Level Security (RLS) ensures users only see their own data // No API route needed - security is in the database!

Step 4: Convert API Routes to Edge Functions

-- ============================================= -- CHANNEL CONNECTIONS -- ============================================= CREATE TABLE inbox_channels ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), channel_type TEXT NOT NULL, -- instagram, whatsapp, email, messenger, sms name TEXT NOT NULL, -- Display name account_id TEXT, -- External account identifier credentials JSONB, -- Encrypted API keys/tokens settings JSONB DEFAULT '{}', is_active BOOLEAN DEFAULT TRUE, last_sync_at TIMESTAMPTZ, created_at TIMESTAMPTZ DEFAULT now() ); -- ============================================= -- CONVERSATIONS -- ============================================= CREATE TABLE inbox_conversations ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), channel_id UUID REFERENCES inbox_channels(id), customer_id UUID REFERENCES customers(id), external_id TEXT, -- Platform's conversation ID contact_identifier TEXT, -- Phone, email, username contact_name TEXT, contact_avatar TEXT, status TEXT DEFAULT 'open' CHECK (status IN ('open', 'pending', 'resolved', 'spam')), priority TEXT DEFAULT 'normal' CHECK (priority IN ('low', 'normal', 'high', 'urgent')), assigned_to UUID REFERENCES staff(id), last_message_at TIMESTAMPTZ, last_message_preview TEXT, unread_count INTEGER DEFAULT 0, is_bot_active BOOLEAN DEFAULT FALSE, metadata JSONB DEFAULT '{}', created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now() ); -- ============================================= -- MESSAGES -- ============================================= CREATE TABLE inbox_messages ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), conversation_id UUID REFERENCES inbox_conversations(id) ON DELETE CASCADE, external_id TEXT, -- Platform's message ID direction TEXT NOT NULL CHECK (direction IN ('inbound', 'outbound')), sender_type TEXT NOT NULL CHECK (sender_type IN ('customer', 'staff', 'bot', 'system')), sender_id UUID, -- staff_id if outbound content_type TEXT DEFAULT 'text', -- text, image, video, audio, file, location, template content TEXT, media_url TEXT, media_metadata JSONB, -- filename, size, mime_type status TEXT DEFAULT 'sent' CHECK (status IN ('pending', 'sent', 'delivered', 'read', 'failed')), error_message TEXT, is_internal_note BOOLEAN DEFAULT FALSE, -- Internal team notes replied_to_id UUID REFERENCES inbox_messages(id), created_at TIMESTAMPTZ DEFAULT now() ); -- ============================================= -- MESSAGE TEMPLATES -- ============================================= CREATE TABLE inbox_templates ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), channel_type TEXT, -- NULL = all channels name TEXT NOT NULL, category TEXT, -- greeting, booking, reminder, followup subject TEXT, -- For email content TEXT NOT NULL, variables TEXT[], -- {{customer_name}}, {{booking_date}}, etc. is_approved BOOLEAN DEFAULT FALSE, -- WhatsApp requires approval external_template_id TEXT, -- WhatsApp template ID created_at TIMESTAMPTZ DEFAULT now() ); -- ============================================= -- CANNED RESPONSES / QUICK REPLIES -- ============================================= CREATE TABLE inbox_canned_responses ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), shortcut TEXT UNIQUE NOT NULL, -- /hello, /hours, /price title TEXT NOT NULL, content TEXT NOT NULL, category TEXT, use_count INTEGER DEFAULT 0, created_by UUID REFERENCES staff(id), created_at TIMESTAMPTZ DEFAULT now() ); -- ============================================= -- AUTOMATION RULES -- ============================================= CREATE TABLE inbox_automations ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, description TEXT, trigger_type TEXT NOT NULL, -- new_conversation, keyword, no_response, booking_created trigger_conditions JSONB, -- {"keywords": ["price", "book"], "channel": "whatsapp"} action_type TEXT NOT NULL, -- auto_reply, assign, tag, notify action_config JSONB, -- {"template_id": "...", "assign_to": "..."} is_active BOOLEAN DEFAULT TRUE, priority INTEGER DEFAULT 0, run_count INTEGER DEFAULT 0, created_at TIMESTAMPTZ DEFAULT now() ); -- Indexes CREATE INDEX idx_conversations_customer ON inbox_conversations(customer_id); CREATE INDEX idx_conversations_channel ON inbox_conversations(channel_id); CREATE INDEX idx_conversations_status ON inbox_conversations(status); CREATE INDEX idx_conversations_assigned ON inbox_conversations(assigned_to); CREATE INDEX idx_conversations_last_msg ON inbox_conversations(last_message_at DESC); CREATE INDEX idx_messages_conversation ON inbox_messages(conversation_id); CREATE INDEX idx_messages_created ON inbox_messages(created_at);
